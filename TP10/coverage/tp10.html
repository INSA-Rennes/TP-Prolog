/**
 * Question 1.1
 * make_all_pairs(+Members, -Pairs)
 */
make_all_pairs([], []) 1 .
make_all_pairs([First|Members], [likes(First, First)|Pairs]) :-
         4 make_pairs(First, Members, Pairs1),
         4 make_all_pairs(Members, Pairs2),
         4 concat(Pairs1, Pairs2, Pairs) 4 .

/**
 * make_pairs(+Member, +Members, -Pairs)
 */
make_pairs(Member, [], []) 4 .
make_pairs(Member, [First|Members], [likes(First, Member), likes(Member, First)|Pairs]) :-
         6 make_pairs(Member, Members, Pairs) 6 .

/**
 * concat(+X, +Y, ?T)
 */
concat([], Y, Y) 4 .
concat([P|R], Y, [P|T]) :-
         12 concat(R, Y, T) 12 .


/**
 * Question 1.2
 * sub_list(+List, -SubList)
 */
sub_list([], []) 65536 .
sub_list([First|List], [First|SubList]) :-
         65535 sub_list(List, SubList) 524288 .
sub_list([First|List], SubList) :-
         65535 sub_list(List, SubList) 524288 .


/**
 * Question 1.3
 * propositionI(+World)
 */
/**
 * Dana likes Cody.
 */
proposition1(World) :-
         65536 member(likes(dana, cody), World) 32768 .

/**
 * Bess does not like Dana.
 */
proposition2(World) :-
         32768 not member(likes(bess, dana), World) 16384  16384 .

/**
 * Cody does not like Abby.
 */
proposition3(World) :-
         16384 not member(likes(cody, abby), World) 8192  8192 .

/**
 * Nobody likes someone who does not like him.
 */
proposition4(World) :-
         8192 not notproposition4(World) 8064  128 .
notproposition4(World) :-
         8192 member(likes(Nobody, Someone), World),
         18752 not member(likes(Someone, Nobody), World) 10688  8064 .

/**
 * Abby likes everyone who likes Bess.
 */
proposition5(World) :-
         128 not notproposition5(World) 96  32 .
notproposition5(World) :-
         128 member(likes(X, bess), World),
         144 not member(likes(abby, X), World) 48  96 .

/**
 * Dana likes everyone Bess likes.
 */
proposition6(World) :-
         32 not notproposition6(World) 12  20 .
notproposition6(World) :-
         32 member(likes(bess, X), World),
         20 not member(likes(dana, X), World) 8  12 .

/**
 * Everybody likes somebody.
 */
proposition7(World) :-
         20 not notproposition7(World) 16  4 .
notproposition7(World) :-
         20 people(People),
         20 member(Somebody, People),
         44 not member(likes(Somebody, _18882), World) 28  16 .


/**
 * propositions(+World)
 */
propositions(World) :-

         32768 proposition2(World),
         16384 proposition3(World),
         8192 proposition4(World),
         128 proposition5(World),
         32 proposition6(World),
         20 proposition7(World) 4 .

         1 people(People),
         1 make_all_pairs(People, Pairs),
         1 sub_list(Pairs, World),
         65536 propositions(World) 4 .
